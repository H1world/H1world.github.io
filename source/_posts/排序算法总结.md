### 冒泡
比较相邻的两个元素，如果前一个比后一个大，则交换位置，比较完一轮之后最后一个将是最大的，再次循环最后一个则不做比较
```js
function bSort(arr) {
  let len = arr.length
  for(let i = 0; i < len-1; i++) {
    for(let j = 0; j < len-1-i; j++){
      if(arr[j] > arr[j+1]){
        let temp = arr[j]
        arr[j] = arr[j+1]
        arr[j+1] = temp
      }
    }
  }
  return arr
};
```
